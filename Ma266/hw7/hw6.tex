\begin{homeworkProblem}
\[
\begin{split}
y'' +y'-2y=2t\\
(r+2)(r-1)=0
\end{split}
\]
So the characteristic equation is 
\[
y=c_1e^{-2t}+c_2e^{t}+Y(t)
\]
Let \(Y(t)= At+B \)
\[
\begin{split}
y'(t)=A \\
y''=0\\
A-2(At+B)=2t\\
A-2B=0\\
-2At=2t\\
A=-1\\
B=-\frac{1}{2}\\
\end{split}
\]
so the formula is 
\[
y=c_1e^{-2t}+c_2e^{t}-t-\frac{1}{2}
\]
Plug in the IV, we get
\[
y=-\frac{1}{2}e^{-2t}+e^{t}-t-\frac{1}{2}
\]
\end{homeworkProblem}

\begin{homeworkProblem}
\[
\begin{split}
y_1'(t)=1\\
y_1''(t)=0\\
\end{split}
\]
Plug it in, we get
\( -(1+t)+1+t=0 \)

\[
\begin{split}
y_2'(t)=e^t\\
y_2''(t)=e^t\\
\end{split}
\]
Plug it in, we get
\(e^tt-(1+t)e^t +e^t=0 \)
So \(y_1\) and \(y_2\) are both the solution of the given equation

\[
y''-\frac{1+t}{t}y'+\frac{1}{t}y=te^{2t}
\]

And the Wronskian of the equation is \(te^t\)

So the general solution will be 
\[
\begin{split}
Y(t)&= -(1+t)\int\frac{e^tte^{2t}}{te^t}+e^t\int\frac{(1+t)te^{2t}}{te^t}\\
&= -(1+t)\int e^{2t} +e^t \int (1+t)e^t  \\
&= -\frac{(1+t)}{2}e^{2t}+e^t(e^t+\int te^t)
\end{split}
\]
Integrate by part, we get
\[
	\int te^tdt= te^t-\int e^tdt=te^t-e^t
\]
Hence 
\[
\begin{split}
-\frac{(1+t)}{2}e^{2t}+e^t(e^t+\int te^t) \\
=-\frac{(1+t)}{2}e^{2t}+te^{2t} \\
=\frac{1}{2}te^{2t}-\frac{1}{2}e^{2t}
\end{split}
\]
\end{homeworkProblem}

\begin{homeworkProblem}
\subsection{L}
\[
\begin{split}
y''+4y=20e^t\\
y'(0)=0 \ y(0)=\\
s^2y(s)-sy(0)-y'(0)+4y(s)=\frac{20}{s-1} \\
y(s)=\frac{20}{(s-1)(s^2+4)}
\end{split}
\]
Find the inverse of \(\La Y(s) \)
\[
\begin{split}
y(s)&=\frac{A}{s-1}+\frac{Bs+C}{s^2+4}\\
&= \frac{4}{s-1}+\frac{-4s-4}{s^2+4}\\
&= \frac{4}{s-1}-\frac{4s}{s^2+4}-\frac{4}{s^2+4}\\
\end{split}
\]
Looking up the table , we get
\[
y(t)=4e^t-4cos(2t)-2sin(2t)
\]
\end{homeworkProblem}
